let button = document.querySelector('button');
let indexToEat = 0;
button.addEventListener('click', function () {
  let items = document.querySelectorAll('li');
  let snack = items[indexToEat];
  snack.innerText = 'The monster ate it!';
  indexToEat = indexToEat + 1;
});
